üíª Sample SQL Queries

Get Customer Order History

SELECT o.order_id, o.total_amount, o.status, o.created_at,
       p.payment_method, p.payment_status
FROM orders o
LEFT JOIN payments p ON o.order_id = p.order_id
WHERE o.customer_id = 1
ORDER BY o.created_at DESC;

Get Menu Items with Average Ratings

SELECT mi.item_id, mi.item_name, mi.price,
       mc.category_name,
       COALESCE(ROUND(AVG(r.rating_value)::numeric, 1), 0) as avg_rating,
       COUNT(r.rating_id) as rating_count
FROM menu_items mi
INNER JOIN menu_category mc ON mi.category_id = mc.category_id
LEFT JOIN ratings r ON mi.item_id = r.item_id
WHERE mi.is_available = true
GROUP BY mi.item_id, mi.item_name, mi.price, mc.category_name
ORDER BY avg_rating DESC;

Get Today's Revenue

SELECT COUNT(order_id) as total_orders,
       COALESCE(SUM(total_amount), 0) as total_revenue
FROM orders
WHERE DATE(created_at) = CURRENT_DATE
AND status = 'completed';
